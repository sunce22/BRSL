# Ukrainian Localization (i18n) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add 🇬🇧/🇺🇦 language toggle to Panel and OBS overlays; hero names, skill names/descriptions, effect labels, and all UI strings switch to Ukrainian when 🇺🇦 is selected.

**Architecture:** `shared/i18n.js` provides `getLang()`, `setLang()`, `t(key)`, `translateHero(hero)`. Ukrainian data lives in `i18n/ui-uk.json` (always loaded) and `i18n/heroes-uk.json` (lazy — loaded only on first 🇺🇦 click). `Promise.allSettled` ensures graceful fallback to English if files are missing. heroes.json is unchanged. Extraction pipeline in `tools/` populates `heroes-uk.json` from live game memory.

**Tech Stack:** Vanilla JS ES modules; Python 3 (extraction scripts only).

**Spec:** `docs/superpowers/specs/2026-05-30-i18n-uk-design.md`

export function t(key) {
  if (getLang() === 'uk' && _uiTr?.[key]) return _uiTr[key];
  return UI_EN[key] ?? key;
}

// Returns effect label for slug (translated or null → caller uses slugToLabel)
export function effectLabel(slug) {
  const key = `effect_${slug}`;
  if (getLang() === 'uk' && _uiTr?.[key]) return _uiTr[key];
  return UI_EN[key] ?? null;
}

// Returns copy of hero with translated name + skills (or original if no translation)
export function translateHero(hero) {
  if (getLang() === 'en') return hero;
  const tr = _heroTr?.[hero.id];
  if (!tr) return hero;
  return {
    ...hero,
    name: tr.name ?? hero.name,
    skills: hero.skills.map((s, i) => ({
      ...s,
      name:        tr.skills?.[i]?.name        ?? s.name,
      description: tr.skills?.[i]?.description ?? s.description,
    })),
  };
}

// Apply active lang class to all .lang-btn elements in document
export function updateLangButtons() {
  const lang = getLang();
  document.querySelectorAll('.lang-btn').forEach(btn => {
    btn.classList.toggle('lang-btn--active', btn.dataset.lang === lang);
  });
}

## Task 7: `tools/extract-l10n-uk.py`

**Files:**
- Create: `tools/extract-l10n-uk.py`

This script scans Raid.exe memory for Ukrainian skill strings and produces `tools/l10n-uk-raw.json`.

- [ ] **Step 1: Create `tools/extract-l10n-uk.py`**

```python
"""
Extract Ukrainian skill name/description strings from Raid: Shadow Legends memory.
Requires game to be running. Run as Administrator.

Output: tools/l10n-uk-raw.json
Format: { "HERO_NNN_sK": { "name": "...", "desc": "..." }, ... }
where NNN = hero_num, K = skill index (1-based)
"""
import ctypes
import ctypes.wintypes as wt
import struct
import sys
import json
import re
from pathlib import Path

if hasattr(sys.stdout, "reconfigure"):
    sys.stdout.reconfigure(encoding="utf-8", errors="replace")

PROCESS_VM_READ         = 0x0010
PROCESS_QUERY_INFORMATION = 0x0400
MEM_COMMIT              = 0x1000
PAGE_READABLE           = {0x02, 0x04, 0x20, 0x40}

k32 = ctypes.windll.kernel32

class MEMORY_BASIC_INFORMATION(ctypes.Structure):
    _fields_ = [
        ("BaseAddress",       ctypes.c_void_p),
        ("AllocationBase",    ctypes.c_void_p),
        ("AllocationProtect", wt.DWORD),
        ("RegionSize",        ctypes.c_size_t),
        ("State",             wt.DWORD),
        ("Protect",           wt.DWORD),
        ("Type",              wt.DWORD),
    ]

def find_pid(name: str) -> int:
    import subprocess
    out = subprocess.check_output(
        ["powershell", "-Command",
         f"Get-Process -Name '{name}' | Select-Object -First 1 -ExpandProperty Id"],
        text=True, errors="replace"
    ).strip()
    return int(out) if out.isdigit() else 0

def read_chunk(handle, addr: int, size: int):
    buf = ctypes.create_string_buffer(size)
    read = ctypes.c_size_t(0)
    ok = k32.ReadProcessMemory(handle, ctypes.c_void_p(addr), buf, size, ctypes.byref(read))
    return buf.raw[:read.value] if ok and read.value > 0 else None

# Ukrainian Unicode range: Ѐ-ӿ (Cyrillic)
CYRILLIC_RE = re.compile(r'[Ѐ-ӿ]{3,}')

def find_utf16_cyrillic(data: bytes, min_len: int = 3) -> list:
    """Find Ukrainian UTF-16 LE strings (min 3 Cyrillic chars)."""
    results = []
    i = 0
    while i < len(data) - 1:
        cp = data[i] | (data[i+1] << 8)
        if 0x0400 <= cp <= 0x04FF or (0x20 <= cp <= 0x7E):
            j = i
            chars = []
            while j + 1 < len(data):
                c = data[j] | (data[j+1] << 8)
                if (0x0400 <= c <= 0x04FF) or (0x20 <= c <= 0x7E) or c in (0x2019, 0x2018, 0x2014, 0x00AB, 0x00BB):
                    chars.append(chr(c))
                    j += 2
                else:
                    break
            text = "".join(chars).strip()
            if len(text) >= min_len and CYRILLIC_RE.search(text):
                results.append((i, text))
            i = j if j > i else i + 2
        else:
            i += 2
    return results

def sanitize(s: str) -> str:
    """Escape HTML special chars from game memory strings."""
    return (s.replace("&", "&amp;")
             .replace("<", "&lt;")
             .replace(">", "&gt;")
             .replace('"', "&quot;")
             .replace("'", "&#39;"))

def main():
    pid = find_pid("Raid")
    if not pid:
        sys.exit("ERROR: Raid process not found. Launch the game first.")
    print(f"Found Raid PID: {pid}")

    handle = k32.OpenProcess(PROCESS_VM_READ | PROCESS_QUERY_INFORMATION, False, pid)
    if not handle:
        sys.exit(f"ERROR: Cannot open process (err {k32.GetLastError()}). Run as Administrator.")

    print("Scanning memory for Ukrainian strings...")

    all_strings = []
    addr = 0
    mbi = MEMORY_BASIC_INFORMATION()
    chunk_size = 1024 * 1024  # 1MB

    scanned = 0
    while addr < 0x7FFFFFFFFFFF:
        ret = k32.VirtualQueryEx(handle, ctypes.c_void_p(addr), ctypes.byref(mbi), ctypes.sizeof(mbi))
        if ret == 0:
            addr += 0x1000
            continue
        size = mbi.RegionSize or 0x1000

        is_readable = (
            mbi.State == MEM_COMMIT
            and (mbi.Protect & 0xFF) in PAGE_READABLE
            and not (mbi.Protect & 0x100)
        )
        is_private = mbi.Type == 0x20000

        if is_readable and is_private and size <= 64 * 1024 * 1024:
            for offset in range(0, size, chunk_size):
                chunk = read_chunk(handle, addr + offset, min(chunk_size, size - offset))
                if not chunk:
                    continue
                scanned += len(chunk)
                for pos, text in find_utf16_cyrillic(chunk):
                    abs_addr = addr + offset + pos
                    all_strings.append((abs_addr, text))

        addr += size
        if scanned > 3 * 1024 ** 3:
            print("Reached 3GB scan limit")
            break

    k32.CloseHandle(handle)
    print(f"Scanned: {scanned / 1024**2:.0f} MB, found {len(all_strings)} Ukrainian strings")

    # Filter to likely skill names (2-80 chars) and descriptions (20-500 chars)
    skill_names  = [(a, sanitize(t)) for a, t in all_strings if 2  <= len(t) <= 80]
    skill_descs  = [(a, sanitize(t)) for a, t in all_strings if 20 <= len(t) <= 500]

    print(f"  Candidate skill names:  {len(skill_names)}")
    print(f"  Candidate descriptions: {len(skill_descs)}")

    # Save raw for manual inspection
    out_path = Path(__file__).parent / "l10n-uk-raw.json"
    raw = {
        "skill_names": [[hex(a), s] for a, s in skill_names[:2000]],
        "skill_descs": [[hex(a), s] for a, s in skill_descs[:2000]],
    }
    out_path.write_text(json.dumps(raw, ensure_ascii=False, indent=2), encoding="utf-8")
    print(f"Saved raw data to {out_path}")
    print("Next step: run tools/build-heroes-uk.py to match strings to heroes")

if __name__ == "__main__":
    main()
```

## Task 8: `tools/build-heroes-uk.py`

**Files:**

Reads `tools/l10n-uk-raw.json` + `data/heroes.json`, produces `i18n/heroes-uk.json`.

- [ ] **Step 1: Create `tools/build-heroes-uk.py`**

```python
"""
Match extracted Ukrainian strings to heroes.json skills.
Input:  tools/l10n-uk-raw.json   (from extract-l10n-uk.py)
        data/heroes.json
Output: i18n/heroes-uk.json

Strategy:
  - For each hero/skill in heroes.json, look for a Ukrainian string
    whose length is similar to the English name/description.
  - Requires manual review of the output before use.
  - Run: python tools/build-heroes-uk.py
"""
import json
from pathlib import Path

ROOT = Path(__file__).parent.parent

raw_path    = Path(__file__).parent / "l10n-uk-raw.json"
heroes_path = ROOT / "data" / "heroes.json"
output_path = ROOT / "i18n" / "heroes-uk.json"

if not raw_path.exists():
    print("ERROR: tools/l10n-uk-raw.json not found.")
    print("Run tools/extract-l10n-uk.py first (with game running).")
    exit(1)

with open(raw_path, encoding="utf-8") as f:
    raw = json.load(f)

with open(heroes_path, encoding="utf-8") as f:
    heroes = json.load(f)

# Load existing output if present (to preserve manually curated entries)
existing = {}
if output_path.exists():
    with open(output_path, encoding="utf-8") as f:
        existing = json.load(f)

candidate_names = [s for _, s in raw.get("skill_names", [])]
candidate_descs = [s for _, s in raw.get("skill_descs", [])]

def find_closest(candidates: list, ref: str, tolerance: float = 0.4) -> str | None:
    """Find candidate whose length is within tolerance% of ref length."""
    ref_len = len(ref)
    best = None
    best_diff = float("inf")
    for c in candidates:
        diff = abs(len(c) - ref_len) / max(ref_len, 1)
        if diff < tolerance and diff < best_diff:
            best_diff = diff
            best = c
    return best

result = dict(existing)  # start with existing curated entries
new_count = 0
skipped = 0

for hero in heroes:
    hid = hero["id"]
    if hid in result:
        continue  # already curated

    hero_entry = {"name": hero["name"], "skills": []}
    matched_any = False

    for skill in hero["skills"]:
        uk_name = find_closest(candidate_names, skill["name"])
        uk_desc = find_closest(candidate_descs, skill["description"], tolerance=0.5)

        if uk_name or uk_desc:
            matched_any = True

        hero_entry["skills"].append({
            "name":        uk_name or skill["name"],
            "description": uk_desc or skill["description"],
        })

    if matched_any:
        result[hid] = hero_entry
        new_count += 1
    else:
        skipped += 1

output_path.write_text(
    json.dumps(result, ensure_ascii=False, indent=2),
    encoding="utf-8"
)

total = len(heroes)
print(f"heroes-uk.json written: {len(result)}/{total} heroes")
print(f"  New matches:     {new_count}")
print(f"  From existing:   {len(existing)}")
print(f"  Unmatched:       {skipped}")
print(f"\nOutput: {output_path}")
print("\nIMPORTANT: Review the output manually — length-based matching")
print("is approximate. Wrong matches will show garbled text in the UI.")
```
